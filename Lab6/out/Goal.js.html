  }

   * Goal Update function

  {
    if(this.alive)
    {
    console.log("Updating Goal");
    }
  }

   * @function update
   */
  draw()
  {
    if(this.alive)
    {
    console.log("Drawing Goal");
    var canvas = document.getElementById('mycanvas');

  }
